    Create the GitHub Pages deployment workflow and adapt smoke tests.

    **deploy.yml** -- GitHub Actions workflow for deploying the Astro site:
    ```yaml
    name: Deploy to GitHub Pages
    on:
      push:
        branches: [main]
      workflow_dispatch:
    permissions:
      contents: read
      pages: write
      id-token: write
    jobs:
      build:
        runs-on: ubuntu-latest
        defaults:
          run:
            working-directory: ./astro-site
        steps:
          - name: Checkout
            uses: actions/checkout@v5
          - name: Install, build, and upload
            uses: withastro/action@v5
            with:
              path: ./astro-site
      deploy:
        needs: build
        runs-on: ubuntu-latest
        environment:
          name: github-pages
          url: ${{ steps.deployment.outputs.page_url }}
        steps:
          - name: Deploy to GitHub Pages
            id: deployment
            uses: actions/deploy-pages@v4
    ```

    Key points:
    - Uses `withastro/action@v5` (official Astro GitHub Action, handles package manager detection)
    - `path: ./astro-site` because Astro project is in a subdirectory
