      hexcode: string;
      hexLower: string;
      group: string;
      subgroup: string;

      useLocalAsset: boolean;
      cldrShortName: string;
      cldrKeywords: string[];
      tags: string[];
      openmoji_tags: string[];
      skintone?: string;
      skintone_combination?: string;
      skintone_base_hexcode?: string;
      unicode?: string;
    }

    export interface Subgroup {
      key: string;
      title: string;
      route: string;         // /{category}/{subgroup}/

      description: string;
      emojis: EmojiEntry[];
    }

    export interface Category {
      key: string;
      title: string;
      route: string;         // /{category}/
      noindex: boolean;
      description: string;
      subgroups: Subgroup[];
    }

    export interface EmojiModel {
      groups: Category[];     // Same as categories for Phase 1
      categories: Category[];
      emojiEntries: EmojiEntry[];
    }

    - Asset paths: `localAssetPath = /assets/emoji/base/{HEX}.svg`, `cdnAssetPath = https://cdn.jsdelivr.net/npm/openmoji@15.1.0/color/svg/{HEX}.svg`
